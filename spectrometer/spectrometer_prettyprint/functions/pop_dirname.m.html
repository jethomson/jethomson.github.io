0001 %POP_DIRNAME - Returns the name of the last directory in dirpath as well as
0002 %              it's parent directory's path.
0003 %
0004 % Syntax:  [dirname, pdirpath] = pop_dirname(dirpath)
0005 %
0006 % Inputs:
0007 %    dirpath  - a path to a directory
0008 %
0009 % Outputs:
0010 %    dirname  - the name of the directory that dir path points to.
0011 %    pdirpath - the path to the parent directory of dirname.
0012 %
0013 %
0014 % Example 1:
0015 %    [dirname, pdirpath] = pop_dirname('path/to/parent/mydirname')
0016 %    -->  dirname = 'mydirname'
0017 %    -->  pdirpath = 'path/to/parent'
0018 %
0019 % Example 2:
0020 % % It's very common for directory paths to be written both with and without
0021 % % a trailing slash. pop_dirname will return the same result whether or not
0022 % % there is a trailing slash. Therefore pop_dirname can be used to make sure
0023 % % a path to a directory is properly formatted.
0024 %    mypath = 'path/to/parent/mydirname'  % correct
0025 %    mypath = 'path/to/parent/mydirname/' % not quite right
0026 %    % OR
0027 %    mypath = 'path/to/parent/mydirname////' % wrong
0028 %    [dirname, pdirpath] = pop_dirname(mypath)
0029 %    s = filesep;
0030 %    goodpath = [pdirpath s dirname]; %no trailing slash
0031 %
0032 % Other m-files required: none
0033 % Subfunctions: none
0034 % MAT-files required: none
0035 %
0036 % See also: none
0037 %
0038 % Author: Jonathan Thomson
0039 % Work:
0040 % email:
0041 % Website: http://jethomson.wordpress.com
0042 %
0043 
0044 function [dirname, pdirpath] = pop_dirname(dirpath)
0045 
0046     if (nargin ~= 1 || isempty(dirpath))
0047         usage('pop_dirname(dirpath)');
0048     end
0049 
0050     s = filesep;
0051     dirname = [];
0052     % looping gets rid of possibly extraneous file separators at the end of
0053     % dirpath
0054     while (isempty(dirname))
0055         [d, dirname] = fileparts(dirpath);
0056         dirpath = d;
0057     end
0058     pdirpath = dirpath;
0059 end
