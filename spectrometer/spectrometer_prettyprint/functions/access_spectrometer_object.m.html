0001 %ACCESS_SPECTROMETER_OBJECT - Controls access to the variables that define the
0002 %                             spectrometer workspace.
0003 %
0004 % The spectrometer object, a structure named 'so', holds the variables camera,
0005 % ftype, exposure, sdate, bd, dir_light, lambda, dlambda, L, Ftri, Hu, and Hr.
0006 % Access to the spectrometer object is not restricted in any way. This accessor
0007 % function was used instead of a global variables because it is more likely to
0008 % prevent the spectrometer object from being accidentally overwritten.
0009 %
0010 % Syntax:  so_out = access_spectrometer_object(so_in)
0011 %
0012 % Inputs:
0013 %     so_in - the spectrometer object as an input.
0014 %
0015 % Outputs:
0016 %    so_out - the stored spectrometer object.
0017 %
0018 % Example:
0019 %    %To set the spectrometer object:
0020 %    access_spectrometer_object(so_in);
0021 %
0022 %    %To get the Bayer filter pattern:
0023 %    so_out = access_spectrometer_out();
0024 %
0025 %
0026 % Other m-files required: none
0027 % Subfunctions: none
0028 % MAT-files required: none
0029 %
0030 % See also: NONE
0031 %
0032 % Author: Jonathan Thomson
0033 % Work:
0034 % email:
0035 % Website: http://jethomson.wordpress.com
0036 %
0037 
0038 function so_out = access_spectrometer_object(so_in)
0039 
0040     if (nargin > 1)
0041         usage('access_spectrometer_object(so_in)');
0042     end
0043 
0044     persistent so;
0045 
0046     if (exist('so_in', 'var') && ~isempty(so_in))
0047         so = so_in;
0048     end
0049 
0050     if (nargout == 1)
0051         so_out = so;
0052     end
0053 
0054 end
