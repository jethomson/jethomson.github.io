0001 %ACCESS_BAYER_PATTERN - Controls access to the Bayer filter pattern.
0002 %
0003 % This function stores the Bayer filter pattern in a persistent variable. Access
0004 % to the filter pattern is not restricted in any way. This accessor function was
0005 % used instead of a global variable because it is more likely to prevent the
0006 % pattern from being accidentally overwritten.
0007 %
0008 % Syntax:  BFP = access_bayer_pattern(pattern)
0009 %
0010 % Inputs:
0011 %    pattern - the Bayer filter pattern.
0012 %
0013 % Outputs:
0014 %    BFP - the stored Bayer filter pattern.
0015 %
0016 % Example:
0017 %    %To set the Bayer filter pattern:
0018 %    access_bayer_pattern(pattern);
0019 %
0020 %    %To get the Bayer filter pattern:
0021 %    BFP = access_bayer_pattern();
0022 %
0023 %
0024 % Other m-files required: none
0025 % Subfunctions: none
0026 % MAT-files required: none
0027 %
0028 % See also: BAYER_DEMOSAIC
0029 %
0030 % Author: Jonathan Thomson
0031 % Work:
0032 % email:
0033 % Website: http://jethomson.wordpress.com
0034 %
0035 
0036 function argout = access_bayer_pattern(pattern)
0037 
0038     if (nargin > 1)
0039         usage('access_bayer_pattern(pattern)');
0040     end
0041 
0042     persistent BFP;
0043 
0044     if (exist('pattern', 'var') && ~isempty(pattern))
0045         BFP = pattern; %Bayer Filter Pattern
0046     end
0047 
0048     if (nargout == 1)
0049         argout = BFP;
0050     end
0051 
0052 end
