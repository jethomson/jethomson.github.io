0001 % This script creates and initializes the variables that construct the path to
0002 % the spectrographs you wish to process.
0003 %
0004 % This script also sets the Bayer filter pattern. Remember that batch_raw_decode
0005 % outputs the Bayer filter pattern of the images it decodes. Keep in mind that
0006 % cropping a Bayer raw image may change which filter pattern should be used to
0007 % demosaic the image.
0008 
0009 % User-defined variables:
0010 %  camera - camera identifier.
0011 %  ftype - type of file to process. JPG and PGM are supported.
0012 %  exposure - the exposure time of the spectrographs. This is used to load the
0013 %             correct set of master calibration frames.
0014 %  sdate - date spectrographs were taken.
0015 
0016 clear access_spectrometer_object
0017 
0018 so_name = 'b';
0019 switch so_name
0020 
0021 case 'a'
0022     so.camera = 'A75';
0023     so.ftype = 'JPG';
0024     so.exposure = '1dot6s';
0025     so.sdate = '2012_08_12';
0026 
0027 case 'b'
0028     so.camera = 'A590';
0029 %    so.ftype = 'JPG';
0030     so.ftype = 'PGM';
0031     so.exposure = '2s';
0032 %    so.sdate = '2012_07_22';
0033     so.sdate = '2012_08_30';
0034 %    so.sdate = '2012_09_13';
0035 
0036     %so.raven = 'Galifianakis';
0037 
0038     % set Bayer filter pattern.
0039     % Bayer filter pattern for a DNG from an A590
0040     so.bfp = 'gbrg';
0041     access_bayer_pattern('gbrg');
0042 
0043 otherwise
0044 
0045 end
0046 
0047 s = filesep;
0048 so.bd = ['data' s so.camera s 'frames']; % base directory
0049 
0050 access_spectrometer_object(so);
